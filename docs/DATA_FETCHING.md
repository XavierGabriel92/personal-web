# Data Fetching

This document describes data fetching patterns using TanStack Query (React Query).

For general architecture patterns, see [ARCHITECTURE.md](../ARCHITECTURE.md).

## TanStack Query (React Query)

The application uses **TanStack Query** for all server state management.

## Query Client Setup

```typescript
// src/routes/__root.tsx
export const queryClient = new QueryClient({});

// Wrapped in root route
<QueryClientProvider client={queryClient}>
  <Outlet />
</QueryClientProvider>
```

## Generated Hooks

API hooks are generated by Kubb from the OpenAPI spec. See [API_CLIENT.md](./API_CLIENT.md) for details.

### Basic Query Hook

```typescript
import { useGetApiRoutines } from "@/gen";

function RoutinesList() {
  const { data, isLoading, error } = useGetApiRoutines();
  
  if (isLoading) return <Spinner />;
  if (error) return <Error message={error.message} />;
  
  return <div>{/* Render routines */}</div>;
}
```

### Query Hook with Options

```typescript
const { data, isLoading, error } = useGetApiRoutines({
  query: {
    staleTime: 5 * 60 * 1000, // 5 minutes
    refetchOnWindowFocus: false,
  },
});
```

## Suspense Queries

For route-level data loading, Suspense queries are used:

```typescript
import { useGetApiRoutineByIdSuspense } from "@/gen";

function RoutineDetail({ routineId }: { routineId: string }) {
  const { data } = useGetApiRoutineByIdSuspense(routineId);
  // Component only renders when data is loaded
  return <div>{data.name}</div>;
}
```

Wrap Suspense queries with Suspense boundary:

```typescript
<Suspense fallback={<Spinner />}>
  <RoutineDetail routineId={routineId} />
</Suspense>
```

## Mutations

Mutations for create/update/delete operations:

```typescript
import { usePostApiRoutineCreate, getApiRoutinesQueryKey } from "@/gen";
import { useQueryClient } from "@tanstack/react-query";
import { toast } from "sonner";

function CreateRoutine() {
  const queryClient = useQueryClient();
  const { mutate, isPending } = usePostApiRoutineCreate({
    mutation: {
      onSuccess: () => {
        // Invalidate and refetch routines list
        queryClient.invalidateQueries({
          queryKey: getApiRoutinesQueryKey(),
        });
        toast.success("Routine created!");
      },
      onError: (error) => {
        toast.error(error.message || "Failed to create routine");
      },
    },
  });
  
  const handleSubmit = (data: RoutineFormData) => {
    mutate({ data });
  };
  
  return (
    <form onSubmit={handleSubmit}>
      {/* Form fields */}
      <Button type="submit" disabled={isPending}>
        {isPending ? "Creating..." : "Create"}
      </Button>
    </form>
  );
}
```

## Optimistic Updates

For better UX, optimistic updates are used:

```typescript
import { useQueryClient } from "@tanstack/react-query";
import { getApiWorkoutByIdSuspenseQueryKey } from "@/gen";

const handleDragExercises = (exercises: WorkoutExercise[]) => {
  const queryKey = getApiWorkoutByIdSuspenseQueryKey(workoutId);
  const oldData = queryClient.getQueryData(queryKey);
  
  // Optimistically update cache
  queryClient.setQueryData(queryKey, (old) => ({
    ...old,
    exercises: exercises,
  }));
  
  // Then call API
  reorderExercises(
    { data: { exerciseWorkouts: exercisesData } },
    {
      onSuccess: (response) => {
        queryClient.setQueryData(queryKey, response);
        toast.success("Order updated!");
      },
      onError: () => {
        // Rollback on error
        queryClient.setQueryData(queryKey, oldData);
        toast.error("Failed to update order");
      },
    }
  );
};
```

## Session Management

Custom hook for session management:

```typescript
// src/hooks/auth.ts
import { queryClient } from "@/routes/__root";
import { useSuspenseQuery } from "@tanstack/react-query";
import { getSession } from "@/lib/auth-client";

export const sessionQueryKey = ["session"];

const fetchSession = async () => {
  const session = await getSession();
  return session.data;
};

export const cachedSession = () => {
  return queryClient.fetchQuery({
    queryKey: sessionQueryKey,
    queryFn: fetchSession,
    staleTime: 5 * 60 * 1000, // 5 minutes
  });
};

export const useCachedSession = () => {
  return useSuspenseQuery({
    queryKey: sessionQueryKey,
    queryFn: fetchSession,
    staleTime: 5 * 60 * 1000,
  });
};
```

## Query Invalidation Patterns

### After Mutations

```typescript
const { mutate } = usePostApiRoutineCreate({
  mutation: {
    onSuccess: () => {
      queryClient.invalidateQueries({
        queryKey: getApiRoutinesQueryKey(),
      });
    },
  },
});
```

### Selective Invalidation

```typescript
// Invalidate specific routine
queryClient.invalidateQueries({
  queryKey: getApiRoutineByIdSuspenseQueryKey(routineId),
});

// Invalidate all routines
queryClient.invalidateQueries({
  queryKey: getApiRoutinesQueryKey(),
});
```

## Error Handling

### Query Errors

```typescript
const { data, error } = useGetApiRoutines();

if (error) {
  // Handle error
  return <ErrorComponent error={error} />;
}
```

### Mutation Errors

```typescript
const { mutate } = usePostApiRoutineCreate({
  mutation: {
    onError: (error) => {
      toast.error(error.message || "An error occurred");
      // Log error, show user-friendly message, etc.
    },
  },
});
```

## Loading States

### Query Loading

```typescript
const { data, isLoading, isFetching } = useGetApiRoutines();

if (isLoading) {
  return <Spinner />; // Initial load
}

if (isFetching) {
  return <div>Refreshing...</div>; // Background refetch
}
```

### Mutation Loading

```typescript
const { mutate, isPending } = usePostApiRoutineCreate();

<Button disabled={isPending}>
  {isPending ? "Creating..." : "Create"}
</Button>
```

## Best Practices

- ✅ Use generated hooks from Kubb
- ✅ Use Suspense queries for route-level data
- ✅ Implement optimistic updates for better UX
- ✅ Invalidate queries after mutations
- ✅ Use query keys consistently
- ✅ Handle loading and error states
- ✅ Use staleTime to reduce unnecessary refetches
- ✅ Rollback optimistic updates on error
